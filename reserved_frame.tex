    \begin{align*}
        \min_{u} \{ \mathcal{J}(u) = \lambda {\color{red} \frac{1}{2} \int_{\Omega} (u-z)^2 d\Omega} +  {\color{blue}\int_{\Omega}  |\nabla u|  d\Omega} \}
    \end{align*}
        $$ \Big \downarrow$$
    \begin{align*}
        \min_{u} \{ \mathcal{J}(u) = \lambda {\color{red} \mathcal{D}(u,z)} +  {\color{blue} \mathcal{R}(u) } \}
    \end{align*}
